Else: Suffix -i

Adjective → verb (to make something [adj]) = Prefix ra-

Noun → adjective relating to noun (e.g. economy → economic) = Suffix -ak
Noun to verb = Suffix -up
Verb → adjective (result of doing [verb]) = If ends with vowel: Suffix -w
Else: Suffix -a
Tending to = If ends with vowel: Suffix -w
Else: Suffix -e

Else: Suffix -e
